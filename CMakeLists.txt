cmake_minimum_required(VERSION 2.6)
project(cppjsconsole)
set(cppjsconsole_VERSION_MAJOR 0)
set(cppjsconsole_VERSION_MINOR 1)
set(cppjsconsole_VERSION_PATCH 0)
include_directories(include/)
add_definitions("-Wall -Wextra -O2")
add_executable(testdemo src/test.cpp src/cppjsconsole.cpp)
install(FILES include/cppjsconsole.h DESTINATION include)
install(FILES src/cppjsconsole.cpp DESTINATION include)
add_library(libcppjsconsole_static STATIC src/cppjsconsole.cpp)
set_target_properties(libcppjsconsole_static PROPERTIES OUTPUT_NAME "libcppjsconsole")
target_link_libraries(testdemo libcppjsconsole_static)
install(TARGETS libcppjsconsole_static ARCHIVE DESTINATION lib)
